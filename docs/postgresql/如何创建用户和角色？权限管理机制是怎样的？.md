```sql
-- 不可登录
create user user_name;

-- 可登录
create user username with password '123444';

CREATE ROLE app_user WITH LOGIN PASSWORD 'secure123' CREATEDB VALID UNTIL '2027-01-01';
CREATE ROLE app_user WITH LOGIN PASSWORD 'secure123' SUPERUSER VALID UNTIL '2027-01-01';

CREATE ROLE developers; -- 组角色
CREATE ROLE alice WITH LOGIN PASSWORD 'alice123';
GRANT developers TO alice;

-- 授予表权限
GRANT SELECT, INSERT ON TABLE my_table TO app_user;

-- 授予整个模式的 USAGE 和 CREATE
GRANT USAGE, CREATE ON SCHEMA public TO developer;

-- 授予数据库连接权限
GRANT CONNECT ON DATABASE mydb TO app_user;

-- 设置当前用户将来创建的表自动授予 SELECT 给 app_reader
ALTER DEFAULT PRIVILEGES
    GRANT SELECT ON TABLES TO app_reader;

-- 回收权限
REVOKE INSERT ON TABLE my_table FROM app_user;
REVOKE ALL PRIVILEGES ON DATABASE mydb FROM app_user;
```
